1.  The implementation of some of the commonly used macros in the
Clojure language

unless:
(defmacro unless [pred a b]
  `(if (not ~pred) ~a ~b))


nif:
(defmacro nif [expr pos zer neg]
	'(case (Integer/signum ~expr)
	-1 ~neg
	0 ~zer
	1 ~pos))

when:
(defmacro when [test &body]
	(list 'if test(cons 'do body)))
